FROM (
    SELECT num
    FROM MyNumbers
    GROUP BY num
    HAVING COUNT(num) = 1
) AS single_numbers;

SELECT employee_id,
       CASE 
           WHEN MOD(employee_id,2) = 1 AND SUBSTR(name,1,1) != 'M' 
           THEN salary 
           ELSE 0 
       END AS bonus
FROM Employees
ORDER BY employee_id;
